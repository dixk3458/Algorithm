function solution(n) {
  const answer = [];
  const ch = Array(n + 1).fill(0);
  function DFS(v) {
    if (v > n) {
      let tmp = '';
      for (let i = 1; i <= n; i++) {
        if (ch[i] === 1) {
          tmp = tmp + String(i) + ' ';
        }
      }

      if (tmp) {
        answer.push(tmp.trim(''));
      }
    } else {
      // ë‘ ê°ˆëž˜ë¡œ ë»—ëŠ”ë‹¤.
      ch[v] = 1;
      DFS(v + 1);
      ch[v] = 0;
      DFS(v + 1);
    }
  }
  DFS(1);

  return answer;
}

console.log(solution(3));
